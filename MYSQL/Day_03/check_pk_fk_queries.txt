1) Not null
-----------------------
the constraints specify that the column can't have a empty or No null Values

create table Sample(eid int , ename varchar(20) not null);
insert into sample(eid,ename) values (101, "Tharun");         ---> Success
insert into sample(eid,ename) values (101, Null);             ---> Error


2) Unique
-----------------------

the Constraints ensures that all values inserted into the column will be Unique

create table Sample(eid int , ename varchar(20) unique ,salary double);


insert into sample(eid,ename,salary) values (101, "Tharun",20000);      -->  Success
insert into sample(eid,ename,salary) values (101, "AMAR",30000);        -->  Success         
insert into sample(eid,ename,salary) values (101, "MANU",50000);        -->  Success
insert into sample(eid,ename,salary) values (101, "kalpana",32000);     -->  Success
insert into sample(eid,ename,salary) values (101, "Tharun",70000);      -->  Error
           
                name tharun is repeated , we will get  Error

3) default
----------------------
pass default value into a Column

create table Sample(eid int , ename varchar(20),ecity varchar(30) default "AP", salary double);


insert into sample(eid,ename,ecity,salary) values (101, "Tharun","TG",20000); 
insert into sample(eid,ename,ecity,salary) values (101, "Tharun",20000);      # default city is "AP"

5) Primary key
-------------------------

primary key Constraints uniquly identifies each record in a table
primary key must contain Unique values and can't conatins Null Values
a table can have only one pK ,  can consists of single or multiple columns


create table covid_cases_wave_1( 
                                id int not null auto_increment,
                                year int ,
                                positive bigint not null, 
                                recovery bigint not null ,
                                dead int     not null,
                                primary key(id)
                               );

desc covid_cases_wave_1;

insert into covid_cases_wave_1(id,year,positive,recovery,dead) 
          values (1,2020,100000,95000,5000);
insert into covid_cases_wave_1(id,year,positive,recovery,dead) 
          values (Null,2024,1000,1000,0);

show tables;
select * from covid_cases_wave_1;




6) Forign Key
------------------------------

forign key is a field ( or Collection of Fields) in one table,
that refers to the PrimaryKey in another table.

the table with the forign key is called the childtable , 
and the table with the Primary key is called the reffered or parent table.

parent table ---> covid_cases_wave_1

create table covid_details_in_AP(
                                 id_1 int not null auto_increment,
                                 all_cases bigint,
                                 val_1 int,
                                 recoverycases bigint,
                                 death_cases bigint,
                                 year int ,
                                 primary key(id_1),
                                 foreign key(val_1) references covid_cases_wave_1 (id)
                                );

create table covid_cases_wave_1( 
                                id int not null auto_increment,
                                year int not null ,
                                positive bigint not null, 
                                recovery bigint not null ,
                                dead int     not null,
                                primary key(id)
                               );

create table covid_details_in_AP(
                                 p_id int not null auto_increment,
                                 all_cases bigint,
                                 recoverycases bigint,
                                 death_cases bigint,
                                 year int,
                                 primary key(p_id),
                                 foreign key (year) references covid_cases_wave_1 (id)
                                );

Auto_increment
----------------------------

Auto_increment allows a unique to be generated automatically ,
when a new record is inserted into a table ,

often this is the primarykey field , that we would like to be created 
by default the starting default value is 1,  and it will increment by 1 for each new record


create table stu_data(id int not null auto_increment,name varchar(10));
